To create a ServiceLevelObjective, apply a resource like:

  apiVersion: observability.slok.io/v1alpha1
  kind: ServiceLevelObjective
  metadata:
    name: my-slo
  spec:
    displayName: "My Application SLO"
    objectives:
      - name: availability
        target: 99.9
        window: 7d
        sli:
          query: |
            sum(rate(http_requests_total{status=~"2.."}[5m]))
            / sum(rate(http_requests_total[5m])) * 100
